
\section {Commandes LATEX spécifiques}

\subsection {Commandes de Texte}

\subsubsection{UKword}

Mise en italique du texte pour les mots anglos axons

\begin{verbatim}
  {\newcommand\UKword[1]{\emph{#1}}}
\end{verbatim}

Exemple :  \verb {\UKword{Texte}} pour :  \UKword{Texte Italique}

\subsubsection{Ucenter}

Centre le texte dans l'environnement

\begin{verbatim}
  \newcommand\Ucenter[1]{\begin{center}{#1}\end{center}}
\end{verbatim}

Exemple : \verb {\Ucenter} \\

\Ucenter {Texte centré dans l'environnement}

\subsubsection{warningbox}

Box avec titre

\begin{verbatim}
\newtcolorbox{warningbox}[2][]
\end{verbatim}

Exemple :  
\begin{verbatim}
 \begin{warningbox} {Titre Box} texte de la box \end {warningbox} 
\end{verbatim} 

\begin{warningbox}{Titre Box} 
texte de la box	
\end{warningbox}
